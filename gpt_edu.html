class Value:
    """自动微分值类 - 就像考试自动批改系统"""
    __slots__ = ('data', 'grad', '_children', '_local_grads')

    def __init__(self, data, children=(), local_grads=()):
        self.data = data           # 当前数值（比如：这道题得分）
        self.grad = 0              # 梯度：这个数值错误对整个考试的影响
        self._children = children  # 子节点：这个数值是怎么计算出来的
        self._local_grads = local_grads  # 本地梯度：子节点对这个数值的贡献比例

    # 各种数学运算，都记录计算关系
    def __add__(self, other):
        other = other if isinstance(other, Value) else Value(other)
        return Value(self.data + other.data, (self, other), (1, 1))

    def __mul__(self, other):
        other = other if isinstance(other, Value) else Value(other)
        return Value(self.data * other.data, (self, other), (other.data, self.data))

    def log(self):
        return Value(math.log(self.data), (self,), (1/self.data,))

    def exp(self):
        return Value(math.exp(self.data), (self,), (math.exp(self.data),))

    def relu(self):
        return Value(max(0, self.data), (self,), (float(self.data > 0),))

    def backward(self):
        """反向传播 - 自动批改系统追责：谁该为错误负责"""
        # 1. 按计算顺序排序（拓扑排序）
        topo = []
        visited = set()

        def build_topo(v):
            if v not in visited:
                visited.add(v)
                for child in v._children:
                    build_topo(child)
                topo.append(v)

        build_topo(self)

        # 2. 从最终扣分开始，反向分摊责任
        self.grad = 1  # 总扣分设为1，然后分摊
        for v in reversed(topo):
            for child, local_grad in zip(v._children, v._local_grads):
                child.grad += local_grad * v.grad

        return self

# 模型超参数（学习能力配置）
N_LAYER = 1      # 大脑思考深度（推理层数）
N_EMBD = 16      # 每个字母的"学习笔记"长度
BLOCK_SIZE = 16  # 最长能记忆的字母数（上下文长度）

HEAD_DIM = N_EMBD // N_HEAD

# 创建随机参数的函数（初始知识是随机的）
def create_matrix(rows, cols, std=0.08):
    return [[Value(random.gauss(0, std)) for _ in range(cols)] for _ in range(rows)]

# 存储所有参数的字典

params['position_embeddings'] = create_matrix(BLOCK_SIZE, N_EMBD)  # 位置嵌入

for layer_idx in range(N_LAYER):
    # 注意力查询、键、值、输出投影

    params[f'layer{layer_idx}.attention.key'] = create_matrix(N_EMBD, N_EMBD)

def linear(x, weights):
    """线性变换：用学到的规则处理信息"""
    # x: 输入向量，weights: 权重矩阵
    output = []
    for row in weights:
        total = sum(w * xi for w, xi in zip(row, x))
        output.append(total)
    return output

def softmax(logits):
    """Softmax：把计算分数转化为选择题概率（每个选项的可能性）"""
    # 1. 减去最大值防止数值溢出
    max_val = max(val.data for val in logits)
    shifted = [l - max_val for l in logits]

    # 2. 计算exp并求和
    exps = [x.exp() for x in shifted]
    total = sum(exps)

    # 3. 归一化
    return [e / total for e in exps]

def rmsnorm(x):
    """RMS归一化：调整思考强度，防止某些想法太强或太弱"""
    # 计算均方值
    mean_square = sum(xi * xi for xi in x) / len(x)

    # 缩放因子
    scale = (mean_square + 1e-5) ** -0.5

    # 应用缩放
    return [xi * scale for xi in x]

def gpt_forward(token_id, position_id, attention_keys, attention_values):
    """
    GPT前向传播：学生完整的思考过程
    输入：当前字母ID、位置ID
    输出：预测下一个字母的概率分布
    """
    # 1. 获取字母的"学习笔记"
    token_embedding = params['token_embeddings'][token_id]

    # 2. 获取位置的"位置笔记"
    position_embedding = params['position_embeddings'][position_id]

    # 3. 合并字母和位置信息
    x = [t + p for t, p in zip(token_embedding, position_embedding)]
    x = rmsnorm(x)  # 归一化：调整思考强度

    # 4. 逐层思考（我们只有1层）
    for layer_idx in range(N_LAYER):
        # 保存原始信息（残差连接：防止丢失重要信息）
        x_residual = x

        # A. 多头注意力：从多个角度参考之前的信息
        x_norm = rmsnorm(x)  # 先归一化

        # 计算查询、键、值
        q = linear(x_norm, params[f'layer{layer_idx}.attention.query'])  # 查询：我想问什么
        k = linear(x_norm, params[f'layer{layer_idx}.attention.key'])    # 键：我之前记住什么
        v = linear(x_norm, params[f'layer{layer_idx}.attention.value'])  # 值：我之前思考什么

        # 存储键值对供后续参考
        attention_keys[layer_idx].append(k)
        attention_values[layer_idx].append(v)

        # 多头注意力计算
        attention_output = []
        for head_idx in range(N_HEAD):
            # 分割成不同的"思考角度"
            head_start = head_idx * HEAD_DIM
            head_end = head_start + HEAD_DIM

            q_head = q[head_start:head_end]
            k_heads = [ki[head_start:head_end] for ki in attention_keys[layer_idx]]
            v_heads = [vi[head_start:head_end] for vi in attention_values[layer_idx]]

            # 注意力分数：当前思考与之前记忆的相似度
            attention_scores = []
            for t in range(len(k_heads)):
                score = sum(q_head[j] * k_heads[t][j] for j in range(HEAD_DIM))
                score = score / math.sqrt(HEAD_DIM)  # 缩放
                attention_scores.append(score)

            # Softmax：决定关注哪些记忆
            attention_weights = softmax(attention_scores)

            # 加权求和：综合参考的记忆
            head_result = []
            for j in range(HEAD_DIM):
                weighted_sum = sum(attention_weights[t] * v_heads[t][j]
                                  for t in range(len(v_heads)))
                head_result.append(weighted_sum)

            attention_output.extend(head_result)

        # 注意力输出投影
        x_attention = linear(attention_output, params[f'layer{layer_idx}.attention.output'])

        # 残差连接：保留原始信息
        x = [a + r for a, r in zip(x_attention, x_residual)]

        # B. MLP层：更深层次的思考
        x_residual = x
        x_norm = rmsnorm(x)
        x_fc1 = linear(x_norm, params[f'layer{layer_idx}.mlp.fc1'])
        x_relu = [xi.relu() for xi in x_fc1]  # 激活函数：过滤负向思考
        x_fc2 = linear(x_relu, params[f'layer{layer_idx}.mlp.fc2'])
        x = [a + b for a, b in zip(x_fc2, x_residual)]  # 再次残差连接

    # 5. 输出预测：最终选择题答案预测
    logits = linear(x, params['lm_head'])
    return logits

print("✅ GPT前向传播函数定义完成！")
print("🧠 这个函数模拟了学生看到一个字母后的完整思考过程")

学生的完整思考流程： 1. **嵌入查找** = 查到当前字母的学习笔记 2. **位置编码** = 记录这是名字的第几个字母 3. **注意力机制** = 快速浏览之前看过的字母，找出相似规律 4. **MLP思考** = 更深层的推理分析 5. **残差连接** = 防止思考过程中丢失原始重要信息 6. **输出预测** = 给出下一个字母的概率分布

n = min(BLOCK_SIZE, len(tokens) - 1)  # 实际处理长度
print(f"🔢 标记序列: {tokens[:n+1]}... (处理前{n}个)")

# 3. 初始化注意力键值存储
keys = [[] for _ in range(N_LAYER)]
values = [[] for _ in range(N_LAYER)]

# 4. 前向传播：学生逐一预测每个字母
all_losses = []
for pos in range(n):
    current_token = tokens[pos]
    target_token = tokens[pos + 1]

    # 学生思考：预测下一个字母
    logits = gpt_forward(current_token, pos, keys, values)
    probs = softmax(logits)

    # 计算扣分：正确选项的概率的负对数
    target_prob = probs[target_token]
    loss = -target_prob.log()
    all_losses.append(loss)

# 5. 平均扣分
if all_losses:
    avg_loss = (1.0 / n) * sum(all_losses)
    print(f"📉 平均扣分（损失）: {avg_loss.data:.4f}")

    # 反向传播：分析错误责任
    avg_loss.backward()
    print("✅ 反向传播完成：已分析出每个参数的责任比例")
else:
    print("⚠️  无法计算损失")
📝 考试比喻解释：
Adam优化器就像是聪明的学习计划： 1. **动量缓存(m)** = 记住最近的学习方向，避免反复振荡 2. **自适应缓存(v)** = 记住不同知识点的调整幅度，给难的知识点更大的调整空间 3. **偏差校正** = 考虑刚开始学习时缓存不准确的情况 这是比简单梯度下降更高效的学习方法。

for param in all_params:
    param.grad = 0

# 重置Adam缓存
m_momentum = [0.0] * len(all_params)
v_adapt = [0.0] * len(all_params)

# 训练循环
losses_history = []
start_time = time.time()

for step in range(TRAIN_STEPS):
    step_start = time.time()

    # 1. 随机选择一个名字
    name_idx = step % len(names)
    current_name = names[name_idx]

    # 2. 转换为标记
    tokens = [BOS_ID] + [char_to_id[ch] for ch in current_name] + [BOS_ID]
    n = min(BLOCK_SIZE, len(tokens) - 1)

    # 3. 前向传播
    keys = [[] for _ in range(N_LAYER)]
    values = [[] for _ in range(N_LAYER)]

    step_losses = []
    for pos in range(n):
        current_token = tokens[pos]
        target_token = tokens[pos + 1]

        logits = gpt_forward(current_token, pos, keys, values)
        probs = softmax(logits)

        target_prob = probs[target_token]
        loss = -target_prob.log()
        step_losses.append(loss)

    # 计算平均损失
    if step_losses:
        avg_loss = (1.0 / n) * sum(step_losses)

        # 4. 反向传播
        avg_loss.backward()

        # 5. Adam优化器更新
        # 学习率衰减：越学越慢
        lr_t = LEARNING_RATE * (1 - step / TRAIN_STEPS)

        for i, param in enumerate(all_params):
            # 更新动量
            m_momentum[i] = BETA1 * m_momentum[i] + (1 - BETA1) * param.grad
            v_adapt[i] = BETA2 * v_adapt[i] + (1 - BETA2) * param.grad * param.grad

            # 偏差校正（训练早期修正）
            m_hat = m_momentum[i] / (1 - BETA1 ** (step + 1))
            v_hat = v_adapt[i] / (1 - BETA2 ** (step + 1))

            # 参数更新
            param.data -= lr_t * m_hat / (math.sqrt(v_hat) + EPSILON)
            param.grad = 0  # 重置梯度

        losses_history.append(avg_loss.data)
        step_time = time.time() - step_start

        # 进度更新
        progress = ((step + 1) / TRAIN_STEPS) * 100
        print(f"🏃 训练进度 {step+1:2d}/{TRAIN_STEPS} | "
              f"扣分: {avg_loss.data:.4f} | "
              f"学习率: {lr_t:.4f} | "
              f"耗时: {step_time:.2f}s", end='\r')

        # 更新进度条
        document.getElementById('progress8').style.width = `${progress}%`

print("🎭 定义推理生成函数：学生期末考试...")

def generate_names(num_names=10, temperature=0.5, max_length=16):
    """
    生成新的名字
    - num_names: 生成数量
    - temperature: 创意温度（0-1，低=保守，高=大胆）
    - max_length: 最大长度
    """
    print(f"🎯 生成配置：{num_names}个名字，温度={temperature}，最大长度={max_length}")
    generated_names = []

    for name_idx in range(num_names):
        # 重置注意力缓存（每生成一个新名字都清空记忆）
        keys = [[] for _ in range(N_LAYER)]
        values = [[] for _ in range(N_LAYER)]

        # 从BOS开始（考试开始信号）
        current_token = BOS_ID
        generated_chars = []

        # 逐个字母生成
        for pos in range(max_length):
            # 学生思考：预测下一个字母
            logits = gpt_forward(current_token, pos, keys, values)

            # 应用温度：调整概率分布
            # 温度高 -> 概率更均匀 -> 更多创意
            # 温度低 -> 概率更集中 -> 更保守
            scaled_logits = [l / temperature for l in logits]
            probs = softmax(scaled_logits)

            # 根据概率随机采样（就像摇骰子，但受概率影响）
            probs_data = [p.data for p in probs]
            next_token = random.choices(range(VOCAB_SIZE), weights=probs_data)[0]

            # 如果生成到BOS，表示"学生认为名字已完成"
            if next_token == BOS_ID:
                break

            # 转换回字符
            char = id_to_char[next_token]
            generated_chars.append(char)
            current_token = next_token

        # 组合成名字
        name = ''.join(generated_chars)
        generated_names.append(name)

        # 进度显示
        print(f"生成中 {name_idx+1}/{num_names}: '{name}'")

    return generated_names

print("✅ 推理生成函数定义完成！")
print("📝 现在我们可以让学生'参加期末考试'，生成全新的名字了")
